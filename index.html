  <script>

    let selectedPet = null;

    let points = 100;

    let level = 1;

    let exp = 0;

    const expPerLevel = 50;


    function selectPet(pet, name, image) {

      if (selectedPet) {

        alert('你已經選擇了一隻寵物，無法更改！');

        return;

      }

      selectedPet = pet;

      document.getElementById('pet').style.backgroundImage = `url(${image})`;

      document.getElementById('pet-selection').style.display = 'none';

      document.getElementById('pet-container').style.display = 'block';

    }


    function interact(action) {

      if (points < 10) {

        alert('點數不足！請等待恢復。');

        return;

      }


      points -= 10;

      exp += 10;


      if (exp >= expPerLevel) {

        level++;

        exp = 0;

        document.getElementById('level').textContent = level;

        alert(`恭喜！你的寵物升到了等級 ${level}！`);

      }

      document.getElementById('points').textContent = points;


      const pet = document.getElementById('pet');

      const status = document.getElementById('pet-status');

      pet.className = 'pet';


      if (action === 'feed') {

        pet.classList.add('bounce');

        status.textContent = '寵物正在吃東西！ 🍎';

        playSound('sound-feed');

      } else if (action === 'play') {

        pet.classList.add('jump');

        status.textContent = '寵物在跳舞！ 🎵';

        playSound('sound-play');

      } else if (action === 'hug') {

        pet.classList.add('zoom');

        status.textContent = '你擁抱了你的寵物！ 🤗';

        playSound('sound-hug');

      } else if (action === 'spin') {

        pet.classList.add('spin');

        status.textContent = '寵物正在旋轉！ 🌀';

        playSound('sound-spin');

      }


      setTimeout(() => {

        pet.className = 'pet';

        status.textContent = '靜止';

      }, 2000);

    }


    function playSound(id) {

      const sound = document.getElementById(id);

      sound.currentTime = 0;

      sound.play();

    }


    setInterval(() => {

      if (points < 100) {

        points += 20;

        if (points > 100) points = 100;

        document.getElementById('points').textContent = points;

      }

    }, 30 * 60 * 1000);

  </script>
